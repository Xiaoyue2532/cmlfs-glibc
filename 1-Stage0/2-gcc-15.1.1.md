# gcc-15.1.1

```shell
# Prepare source
wget https://ipv6.mirrors.ustc.edu.cn/gnu/mpfr/mpfr-4.2.2.tar.bz2 && \
    tar -xf mpfr-4.2.2.tar.bz2 && \
    rm mpfr-4.2.2.tar.bz2 && \
    ln -sf mpfr-4.2.2 mpfr

wget https://ipv6.mirrors.ustc.edu.cn/gnu/mpc/mpc-1.3.1.tar.gz && \
    tar -xf mpc-1.3.1.tar.gz && \
    rm mpc-1.3.1.tar.gz && \
    ln -sf mpc-1.3.1 mpc

wget https://ipv6.mirrors.ustc.edu.cn/gnu/gmp/gmp-6.3.0.tar.bz2 && \
    tar -xf gmp-6.3.0.tar.bz2 && \
    rm gmp-6.3.0.tar.bz2 && \
    ln -sf gmp-6.3.0 gmp

sed -e '/m64=/s/lib64/lib/' \ -i.orig gcc/config/i386/t-linux64

mkdir build-gcc1 && cd build-gcc1

# Add --disable-bootstrap
../configure \
    --prefix=/lfs/tools \
    --target=x86_64-pc-linux-gnu \
    --with-glibc-version=2.41 \
    --enable-languages=c,c++ \
    --disable-libquadmath \
    --enable-default-pie \
    --enable-default-ssp \
    --disable-libstdcxx \
    --disable-bootstrap \
    --with-sysroot=/lfs \
    --disable-libatomic \
    --disable-multilib \
    --disable-threads \
    --without-headers \
    --disable-libgomp \
    --disable-shared \
    --disable-libssp \
    --disable-libvtv \
    --with-newlib \
    --disable-nls \
    CC=x86_64-pc-linux-gnu-clang \
    CXX=x86_64-pc-linux-gnu-clang++ \
    OBJDUMP=llvm-objdump \
    OBJCOPY=llvm-objcopy \
    READELF=llvm-readelf \
    RANLIB=llvm-ranlib \
    STRIP=llvm-strip \
    SIZE=llvm-size \
    NM=llvm-nm \
    AR=llvm-ar \
    AS=llvm-as \
    LD=ld.lld
```
